122	def initialiser_composant(self, 
123						      nom, 
124						      texte: str, 
125						      valeur_defaut: int, 
126						      unite : str, 
127						      classe : type[QWidget], 
128						      tooltip : str, 
129						      instantane : bool = False) -> None:
130		"""
131		Fonction auxiliaire pour initialiser un composant spécifique.
132		
133		:param nom: Nom du composant
134		:param texte: Texte affichée sur le label du composant
135		:type texte: str
136		:param valeur_defaut: Valeur par défaut du composant
137		:type valeur_defaut: int
138		:param unite: Unité à afficher
139		:type unite: str
140		:param classe: Classe du composant
141		:type classe: type[QWidget]
142		:param tooltip: Texte de l'infobulle
143		:type tooltip: str
144		:param instantane: Si la modification du composant prends effet instantanément
145		:type instantane: bool
146		"""
147		composant = QWidget()
148		arrangement_principal = QVBoxLayout()
149		arrangement_label_icone = QHBoxLayout()
150		champ = classe(self, texte)
151		if not (isinstance(champ, Champ_immunite) or isinstance(champ, Champ_voir_morts)):
152			label = QLabel(f"{texte} : {valeur_defaut}{unite}")
153			setattr(self, f"label_{nom}", label)
154			arrangement_label_icone.addWidget(label)
155			icone = QLabel()
156			pix = QPixmap(":/icons/tooltip.png").scaled(
157				16, 16, Qt.KeepAspectRatio, Qt.SmoothTransformation
158			)
159			icone.setPixmap(pix)
160			arrangement_label_icone.addStretch(1)
161			arrangement_label_icone.addWidget(icone)
162			arrangement_principal.addLayout(arrangement_label_icone)
163		setattr(self, f"champ_{nom}", champ)
164		arrangement_principal.addWidget(champ)
165		
166		if isinstance(champ, Champ_temps_guerison) or isinstance(champ, Champ_distance_infection):
167			getattr(self, f"champ_{nom}").changer_valeur(valeur_defaut)
168			champ.currentIndexChanged.connect(champ.changement_valeur)
169			icone.setToolTip(tooltip)
170		elif isinstance(champ, Champ_immunite) or isinstance(champ, Champ_voir_morts):
171			getattr(self, f"champ_{nom}").changer_valeur(valeur_defaut)
172		else:
173			getattr(self, f"champ_{nom}").setValue(valeur_defaut)
174			champ.valueChanged.connect(champ.changement_valeur)
175			icone.setToolTip(tooltip)
176		composant.setLayout(arrangement_principal)
177		if instantane != 0:
178			self.layout_instantane.addWidget(composant)
179		else:
180			self.layout_non_instantane.addWidget(composant)

def initialiser_composant( self, nom, texte: str, valeur_defaut: int, unite: str, classe: type[PySide6.QtWidgets.QWidget], tooltip: str, instantane: bool = False) -> None: 
123	def initialiser_composant(self, 
124						      nom, 
125						      texte: str, 
126						      valeur_defaut: int, 
127						      unite : str, 
128						      classe : type[QWidget], 
129						      tooltip : str, 
130						      instantane : bool = False) -> None:
131		"""
132		Fonction auxiliaire pour initialiser un composant spécifique.
133		
134		:param nom: Nom du composant
135		:param texte: Texte affichée sur le label du composant
136		:type texte: str
137		:param valeur_defaut: Valeur par défaut du composant
138		:type valeur_defaut: int
139		:param unite: Unité à afficher
140		:type unite: str
141		:param classe: Classe du composant
142		:type classe: type[QWidget]
143		:param tooltip: Texte de l'infobulle
144		:type tooltip: str
145		:param instantane: Si la modification du composant prends effet instantanément
146		:type instantane: bool
147		"""
148		composant = QWidget()
149		arrangement_principal = QVBoxLayout()
150		arrangement_label_icone = QHBoxLayout()
151		champ = classe(self, texte)
152		if not (isinstance(champ, Champ_immunite) or isinstance(champ, Champ_voir_morts)):
153			label = QLabel(f"{texte} : {valeur_defaut}{unite}")
154			setattr(self, f"label_{nom}", label)
155			arrangement_label_icone.addWidget(label)
156			icone = QLabel()
157			pix = QPixmap(":/icons/tooltip.png").scaled(
158				16, 16, Qt.KeepAspectRatio, Qt.SmoothTransformation
159			)
160			icone.setPixmap(pix)
161			arrangement_label_icone.addStretch(1)
162			arrangement_label_icone.addWidget(icone)
163			arrangement_principal.addLayout(arrangement_label_icone)
164		setattr(self, f"champ_{nom}", champ)
165		arrangement_principal.addWidget(champ)
166		
167		if isinstance(champ, Champ_temps_guerison) or isinstance(champ, Champ_distance_infection):
168			getattr(self, f"champ_{nom}").changer_valeur(valeur_defaut)
169			champ.currentIndexChanged.connect(champ.changement_valeur)
170			icone.setToolTip(tooltip)
171		elif isinstance(champ, Champ_immunite) or isinstance(champ, Champ_voir_morts):
172			getattr(self, f"champ_{nom}").changer_valeur(valeur_defaut)
173		else:
174			getattr(self, f"champ_{nom}").setValue(valeur_defaut)
175			champ.valueChanged.connect(champ.changement_valeur)
176			icone.setToolTip(tooltip)
177		composant.setLayout(arrangement_principal)
178		if instantane != 0:
179			self.layout_instantane.addWidget(composant)
180		else:
181			self.layout_non_instantane.addWidget(composant)
